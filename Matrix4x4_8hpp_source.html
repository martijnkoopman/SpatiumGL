302  Matrix<T, 4, 4> inverse() const
303  {
304  const T det = determinant();
305  if (det == 0) {
306  throw std::out_of_range("Matrix has no inverse (matrix is singular)");
307  }
308 
309  Matrix<T, 4, 4> result;
310  for (size_t j = 0; j < 4; j++) {
311  for (size_t i = 0; i < 4; i++) {
312  Matrix<T, 3, 3> minor = omit(i, j);
313  const T r = minor.determinant() / det;
314  result[j][i] = r;
315  if ((i + j) % 2 == 1) {
316  result[j][i] = -result[j][i];
317  }
318  }
319  }
320  return result;

514  static Matrix<T, 4, 4> rotationAround(const Vector<T, 3>& axis, T angle)
515  {
516  Matrix<T, 4, 4> result;
517 
518  const Vector<T, 3> axisNormalized = axis.normalized();
519  const T q0 = cos(angle / 2);
520  const T q1 = sin(angle / 2) * axisNormalized.x();
521  const T q2 = sin(angle / 2) * axisNormalized.y();
522  const T q3 = sin(angle / 2) * axisNormalized.z();
523 
524  // Column 0
525  result[0][0] = q0 * q0 + q1 * q1 - q2 * q2 - q3 * q3;
526  result[0][1] = 2 * (q2 * q1 + q0 * q3);
527  result[0][2] = 2 * (q3 * q1 - q0 * q2);
528  result[0][3] = 0;
529 
530  // Column 1
531  result[1][0] = 2 * (q1 * q2 - q0 * q3);
532  result[1][1] = (q0 * q0 - q1 * q1 + q2 * q2 - q3 * q3);
533  result[1][2] = 2 * (q3 * q2 + q0 * q1);
534  result[1][3] = 0;

536  // Column 2

538  result[2][1] = 2 * (q2 * q3 - q0 * q1);
539  result[2][2] = (q0 * q0 - q1 * q1 - q2 * q2 + q3 * q3);
