277  static BoundingBox fromPoints(const std::vector<Vector<T, 3>>& points)
278  {
279  if (points.size() == 0) {
280  return BoundingBox();
281  }
282 
283  // Set initial bounds
284  Vector<T, 3> minVal(points[0][0], points[0][1], points[0][2]);
285  Vector<T, 3> maxVal(points[0][0], points[0][1], points[0][2]);
286 
287  // Iterate points and update bounds
288  for (size_t i = 1; i < points.size(); i++) {
289  Vector<T, 3> point = points[i];
290 
291  // Update min and max
292  for (size_t i = 0; i < 3; i++) {
293  if (point[i] < minVal[i]) {
294  minVal[i] = point[i];
295  } else if (point[i] > maxVal[i]) {
296  maxVal[i] = point[i];
297  }
298  }
299  }

301  return BoundingBox::fromMinMax(minVal, maxVal);

304  static BoundingBox fromMinMax(const Vector<T, 3>& min,
305  const Vector<T, 3>& max)
306  {
307  // Compute center and radii
308  Vector<T, 3> center(max[0] / static_cast<T>(2) + min[0] / static_cast<T>(2),
309  max[1] / static_cast<T>(2) + min[1] / static_cast<T>(2),
310  max[2] / static_cast<T>(2) +
311  min[2] / static_cast<T>(2));
312 
313  Vector<T, 3> radii(max[0] / static_cast<T>(2) - min[0] / static_cast<T>(2),
314  max[1] / static_cast<T>(2) - min[1] / static_cast<T>(2),
315  max[2] / static_cast<T>(2) - min[2] / static_cast<T>(2));
316 
317  return BoundingBox(center, radii);
318  }
